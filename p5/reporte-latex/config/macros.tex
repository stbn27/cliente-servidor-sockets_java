% Notas
\newcommand{\nota}[3]{%
  \begin{tcolorbox}[
    enhanced,
    breakable,
    colframe=#1!70!black,      % borde: color fuerte
    colbacktitle=#1!80!black,  % fondo del título: color fuerte
    colback=#1!8,              % fondo del cuerpo: color muy suave
    coltitle=white,
    fonttitle=\bfseries,
    title=#2,
    boxrule=0.8pt,
    arc=2pt,
    left=6pt, right=6pt, top=5pt, bottom=5pt
  ]
    #3
  \end{tcolorbox}
}

% \nota{blue}{Exclusión mutua}{%
% La exclusión mutua garantiza que, en un instante dado, \textbf{como máximo un proceso}
% se encuentre dentro de la sección crítica y utilice el recurso compartido.
% }
